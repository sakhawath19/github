##### Git help
git help -a					--- print all available commands in standard window
git help -i 					--- Display manual page for the command in the info format
git help -g					--- display useful guides
git <help command> -h				--- print help in command window
git help <help command>				--- shows help in web browser




###### Git initializtion 
git init <newProject> 				--- it will intialize git with the directory newProject
cd <newProject>					--- move to the newProject directory
git notepad <file1.txt>				--- open a text file 
git add <filename>				--- add updated file or new file
git add <newFolder/newFile.txt> 		--- add a new folder including a file to local repo
git commit -m <"New message">			--- commit with a meaning message 
git push 					--- uploading to remote repo




##### Git initialization removal 
rd /s /q .git 					---Remove git initialization from local repo 




##### Remote repo
git remote 					--- list all the remote connection 
git remote -v 					--- shows remote repository with url
git remote add <name> <url>			--- add remote repo with the given name 
git remote rm <name>				--- remove remote repo 
git remote rename <old-name> <new-name>




###### Add existing project to remote repository 
First create the remote repo such as "clone_test"
Then you will get the url such as "https://github.com/sakhawath19/clone_test.git"

git remote 					--- list all the remote connection 
git remote -v 					--- shows remote repository with url

git remote add origin <remote url> 		--- add existing remote repository
git push -u origin master 			--- upload file to the remote repository

git push <remote url> <branch>			--- creates a local branch in the destination repository




##### Git branch 
git branch --merged 				--- shows which branch has been merged with the current branch
git branch --no--merged 			--- shows which branch has not been merged yet

git push -u origin <branch name> 		--- to upload NewBranch to remote directory

git branch -d <branch name> 			--- delete branch from local
git branch -D <branch name> 			--- forcefully deleting branch without merging 
git push origin --delete <branch name> 		--- delete branch from remote

git branch -v 					--- presents last commit of each branch

git branch -b iss53 				--- creating new branch and checkout at the same time, Working with 53 number issue

git commit -a -m "file3 modified" 		--- direct commit without adding the changed file




##### Git fetch, pull, clone, merge
git fetch <remote> 				--- download all the data from remote repository
git fetch origin 				--- fetch new data which have been modified since you have cloned it
git clone 					--- set a local master branch to track your remote master branch
git clone <remote url>
git pull 					--- download and merge with your existing directory

git push -u origin <branch name> 		--- to upload NewBranch to remote directory. Do it when your local repo tracking the remote repo
git push origin <branch name> 			--- Pushes the changes in your local repository up to the remote repository you specified as the origin

git push <remote url> master 			--- you are uploading your master branch to the given url
git push <remote url> <branch name> 		--- uploading new branch to the remote repo 

git push origin v1.5 				--- transfer tag to remote server




##### History of previous commit
git log
git log --stat
git log --pretty=oneline
git log --all --decorate --oneline --graph 	--- only commit with tracking history will show

git branch -v 					--- to see the last commit of each branch

git remote -v 					--- to see remote repository url

git status -s 					--- short form of status

git show <version>




##### See changes of the updated file 
git diff <filename>				--- shows latest update 
git log --follow -p -- <file name>		--- shows entire history
gitk <filename>					--- to see the changes log
git blame <filename>				--- who have change what in a file




##### Remove file and folder
git rm <file name> 				--- remove file, be careful, it will delete from hard drive
git rm -f <file name> 				--- remove file forcefully, be careful, it will delete from hard drive
git rm -r <folder name> 			--- remove folder and file in it
git rm --cached <file name> 			--- delete file from repo, but remain in the file system or hard drive 
git rm -r --cached 				--- remove everything from repo. But commit everything before doing it
git rm log/\*.log 				--- remove all files that have the .log extension in the log/ directory




##### Commit (find difference between two commit)
git diff <commit-no-1> <commit-no-2>		--- shows difference between two commit
git diff --cached




##### Commit (roll back to previous commit)
git revert --no-commit 0766c053..HEAD 		--- specifi commit id:0766c053
git commit -m <message> 			--- create new commit from the existing commit; No information will be lost but previous state is regained 




##### Git ignore 
# ignore all .a files
*.a

# but do track lib.a, even though you're ignoring .a files above
!lib.a

# only ignore the TODO file in the current directory, not subdir/TODO
/TODO

# ignore all files in any directory named build
build/

# ignore doc/notes.txt, but not doc/server/arch.txt
doc/*.txt

# ignore all .pdf files in the doc/ directory and any of its subdirectories
doc/**/*.pdf 




##### Creating version or tag
git tag -a v0.1 -m "Version v0.1" 		--- tagging a version with your project 
git tag -l 					--- show the list of your version
git tag -1 -l "v1.8.5*" 			--- looking at specific series of version

git tag v1.4-lw 				--- lighweight tag(read more later)

git push origin v0.1 				--- uploading to remote server version 0.1
git push origin tags 				--- pushes all of your tags

git tag -d <version>
git push origin --delete <tagname> 		--- to delete remote version

git show <tag>
